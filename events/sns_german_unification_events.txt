namespace = sns_german_unification_events

country_event = {
	id = sns_german_unification_events.1
	title = sns_german_unification_events.1.t
	desc = sns_german_unification_events.1.d
	
	fire_only_once = yes
	
	trigger = {
		culture_group = germanic
		capital_scope = {
			region = north_german_region
		}
		is_year = 1867
		num_of_owned_provinces_with = {
			value = 20
			region = north_german_region
			culture_group = germanic
		}
		any_country = {
			culture_group = germanic
			capital_scope = {
				region = north_german_region
			}
			NOT = { tag = ROOT }
		}
	}
	
	option = {
		name = sns_german_unification_events.1.a
		
		ai_chance = { factor = 80 }
		
		every_country = {
			limit = {
				culture_group = germanic
				capital_scope = {
					region = north_german_region
				}
				NOT = { tag = ROOT }
			}
			country_event = { id = sns_german_unification_events.2 }
		}
	}
	
	option = {
		name = sns_german_unification_events.1.b
		
		ai_chance = { factor = 20 }
	}
}

				NOT = { is_core = FROM }

		}
	}
}
