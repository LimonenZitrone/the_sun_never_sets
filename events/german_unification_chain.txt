country_event = {
	id = unite.1
	title = unite.1.t
	desc = unite.1.d
	
	fire_only_once = yes
	
	trigger = {
		capital_scope = {
			region = north_german_region
			}
		year = 1867
		num_of_owned_provinces_with = {
			value = 20
			culture_group = germanic
			owned_by = ROOT
			region = north_german_region
			is_overseas = no
			}
		culture_group = germanic
		}
		
	option = {
		name = unite.1.a
		
		every_country = {
			limit = {
				culture_group = germanic
				capital_scope = {
					region = north_german_region
					}
				NOT = {
					tag = ROOT
					}
				}
			country_event = { id = unite.2 }
			}
		}
		
	option = {
		name = unite.1.b
	}
}
